78  int32_t index,
79  const void *src,
80  int bigEndian)
81  {
82  int32_t result = 0L;
83  uint8_t *buffer = (uint8_t *)src;
84  if (bigEndian != 0)
85  {
86  result = (buffer[index + 0] & 0xffL);
